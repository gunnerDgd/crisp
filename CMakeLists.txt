cmake_minimum_required(VERSION 3.0.0)
set     (CMAKE_C_STANDARD 11)
project (atomic-structure VERSION 0.1.0 LANGUAGES C)

option(FEATURES_ATOMIC_STACK OFF)
option(FEATURES_ATOMIC_SPSC  OFF)
option(FEATURES_ATOMIC_MPMC  OFF)

option(BUILD_STATIC_LIBRARY OFF)
option(BUILD_SHARED_LIBRARY ON)
option(BUILD_GDB_SYMBOL     OFF)
option(BUILD_DEBUG_MESSAGE  ON)
option(BUILD_TESTCASE       OFF)

if(UNIX AND NOT APPLE)
        message(STATUS "[PRE-BUILD][ENVIRONMENT] Linux Build System Detected.")
        add_compile_definitions(ATOMIC_STRUCTURE_BUILD_ENVIRONMENT_LINUX)
elseif(WIN32)
        message(STATUS "[PRE-BUILD][ENVIRONMENT] Windows Build System Detected.")
        add_compile_definitions(ATOMIC_STRUCTURE_BUILD_ENVIRONMENT_WINDOWS)
endif()

if("${CMAKE_C_COMPILER_ID}" STREQUAL "GNU")
    message(STATUS "[PRE-BUILD][ENVIRONMENT] GNU Build System Detected.")
    add_compile_definitions(ATOMIC_STRUCTURE_BUILD_ENVIRONMENT_GNU)
    if(BUILD_GDB_SYMBOL)
        add_compile_options(-ggdb)
    endif()
elseif("${CMAKE_C_COMPILER_ID}" STREQUAL "MSVC")
    message(STATUS "[PRE-BUILD][ENVIRONMENT] MSVC Build System Detected.")
    add_compile_definitions(ATOMIC_STRUCTURE_BUILD_ENVIRONMENT_MSVC)
endif()

if(BUILD_DEBUG_MESSAGE)
    add_compile_definitions(ATOMIC_STRUCTURE_DEBUG_MESSAGE)
endif()

add_subdirectory(source/details)
add_subdirectory(source/allocator)

if(FEATURES_ATOMIC_STACK)
    message(STATUS "[PRE-BUILD][FEATURES] atomic_stack")
    add_subdirectory(source/stack)
endif()

if(FEATURES_ATOMIC_SPSC)
    message(STATUS "[PRE-BUILD][FEATURES] atomic_spsc")
    add_subdirectory(source/spsc)
endif()

if(FEATURES_ATOMIC_MPMC)
    message(STATUS "[PRE-BUILD][FEATURES] atomic_mpmc")
    add_subdirectory(source/mpmc)
endif()

if(BUILD_TESTCASE)
    add_subdirectory(test_case)
endif()
